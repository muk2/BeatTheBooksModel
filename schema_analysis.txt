Schema Overview
coaches
Primary key: id

Unique constraint: (season, coach_name, team_abbr)

Columns: season, coach_name, team_abbr, reg/post season stats (games, wins, losses, ties)

Purpose: Tracks coaching records by season and team.

players
Primary key: id

Columns: player_name, age, position, games, games_started

Purpose: Stores player roster and basic stats.

team_games
Primary key: id

Unique constraint: (team_abbr, season, week)

Columns: team_abbr, season, week, day, game_date, game_time, winner/loser, points, yards, turnovers

Purpose: Records individual game results and stats.

team_standings
Primary key: id

Unique constraint: (season, team_name)

Columns: season, conference, division, team_name, playoff/division flags, win-loss-tie record, percentages, point differential, strength of schedule, rating system metrics

Purpose: Tracks overall team performance and standings.
